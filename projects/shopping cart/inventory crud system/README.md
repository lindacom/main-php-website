$result = mysqli_query($connect, $sql);
$productCount = mysqli_num_rows($result); // count the output amount
if ($productCount > 0) {
	while($row = mysqli_fetch_array($result)){ 
             $id = $row["id"];
			 $title = $row["title"];

	$product_list = "You have no products listed in your store yet";
}
